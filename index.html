        function renderPayPalButtons() {
            setTimeout(() => {
                if (window.paypal && paypal.Buttons) {
                    console.log("PayPal SDK rilevato. Renderizzazione pulsante...");
                    paypal.Buttons({
                        fundingSource: paypal.FUNDING.CARD, // Forza la visualizzazione delle carte di credito, incluso AMEX
                        style: {
                            layout: 'vertical',
                            color: 'blue',
                            shape: 'rect',
                            label: 'pay'
                        },
                        createOrder: function(data, actions) {
                            return actions.order.create({
                                purchase_units: [{
                                    amount: {
                                        value: '10.00'
                                    }
                                }]
                            });
                        },
                        onApprove: function(data, actions) {
                            return actions.order.capture().then(function(details) {
                                alert('Pagamento completato con successo da ' + details.payer.name.given_name);
                            });
                        }
                    }).render('#paypal-button-container');
                } else {
                    console.error("PayPal SDK non caricato.");
                }
            }, 3000);
        }

        document.addEventListener("DOMContentLoaded", renderPayPalButtons);
